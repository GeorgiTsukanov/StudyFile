function find_roots_and_plot()
    % Функция для нахождения корней уравнения -x = 0 и построения графиков
    
    % Определяем функцию
    f = @(x) -x;
    
    % Находим корень уравнения f(x) = 0
    root = fzero(f, 0); % Начальное приближение - 0
    
    % Выводим информацию о корне
    fprintf('Корень уравнения -x = 0: x = %.2f\n', root);
    
    % Создаем диапазон значений x для построения графика
    x = linspace(-5, 5, 1000);
    
    % Вычисляем соответствующие значения y
    y = f(x);
    
    % Создаем новое графическое окно
    figure;
    
    % Построение графика функции
    plot(x, y, 'b-', 'LineWidth', 2);
    hold on;
    
    % Отмечаем корень на графике
    plot(root, f(root), 'ro', 'MarkerSize', 10, 'MarkerFaceColor', 'r');
    
    % Добавляем линии осей
    xline(0, 'k-', 'LineWidth', 1); % Ось Y
    yline(0, 'k-', 'LineWidth', 1); % Ось X
    
    % Настройки графика
    grid on;
    title('График функции f(x) = -x');
    xlabel('x');
    ylabel('f(x)');
    legend('f(x) = -x', 'Корень уравнения', 'Location', 'northwest');
    
    % Сообщение о завершении
    disp('График построен. Корень отмечен красной точкой.');
end
